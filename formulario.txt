CREATE TYPE TIPO_ANIMAL AS ENUM ('GATO', 'CACHORRO') 
CREATE TYPE SEXO_ANIMAL AS ENUM ('FEMININO', 'MASCULINO') 

create table pet(
    id_pet SERIAL PRIMARY KEY,
    nome VARCHAR(50) NOT NULL,
    sobrenome VARCHAR(100) NOT NULL,
    tipo VARCHAR(20) NOT NULL CHECK (tipo IN ('GATO', 'CACHORRO')),
    sexo VARCHAR(20) NOT NULL CHECK (sexo in ('FEMININO', 'MASCULINO')),
    idade DECIMAL(4, 2) NOT NULL,
    peso DECIMAL(4,2) NOT NULL,
    cidade VARCHAR(100) NOT NULL,
    bairro VARCHAR(100) NOT NULL,
    rua VARCHAR(100) NOT NULL,
    numero VARCHAR(10) NOT NULL
    data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
